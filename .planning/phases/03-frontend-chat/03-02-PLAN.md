  <action>
Rewrite `apps/frontend/src/App.tsx` to:

1. Import `useSession` from `hooks/useSession`
2. Import `useStreamingMessages` from `hooks/useStreamingMessages`
3. Call `useSession()` → get `sessions`, `currentSession`, `messages` (sessionMessages), `loading`, etc.
4. Call `useStreamingMessages(sessionMessages)` → get `messages`, `isStreaming`, `error`, `sendMessage`, `abort`, `clearError`
5. Load sessions on mount via `useEffect` calling `loadSessions()`
6. Create `handleSend` that calls `sendMessage({ sessionId: currentSession.id, message, mode: currentSession.mode })`
7. Render `ChatShell` passing all required props

```tsx
const {
  sessions, currentSession, messages: sessionMessages, loading,
  error: sessionsError, loadSessions, switchSession, newSession,
  removeSession, setCurrentMode,
} = useSession();

const {
  messages, isStreaming, error: chatError,
  sendMessage, abort, clearError,
} = useStreamingMessages(sessionMessages);

useEffect(() => { loadSessions(); }, [loadSessions]);

const handleSend = (message: string) => {
  if (!currentSession) return;
  sendMessage({ sessionId: currentSession.id, message, mode: currentSession.mode });
};

return (
  <ChatShell
    messages={messages}
    isStreaming={isStreaming}
    onSend={handleSend}
    onAbort={abort}
    sessions={sessions}
    currentSession={currentSession}
    onNewSession={() => newSession()}
    onSelectSession={(id) => switchSession(id)}
    onDeleteSession={(id) => removeSession(id)}
    currentMode={currentSession?.mode ?? 'normal'}
    onModeChange={(mode) => setCurrentMode(mode)}
    error={chatError ?? sessionsError}
    onClearError={clearError}
    loading={loading}
  />
);
```

Wrap all handlers in `useCallback`. The `loading` prop is passed to ChatShell so the sidebar can show a loading state.
