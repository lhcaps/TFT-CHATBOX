```python
async def generate_embeddings(self, texts: list[str]) -> list[list[float]]:

    async with httpx.AsyncClient(timeout=120.0) as client:
        payload = {
            "model": self.embedding_model,
            "input": texts,
            "truncate": True,
            "options": {"num_ctx": 8192},

        response = await client.post(
            f"{self.base_url}/api/embed",
            json=payload,
            timeout=120.0,
        )
        response.raise_for_status()
        data = response.json()
        embeddings = data.get("embeddings", [])
        if not isinstance(embeddings, list):
            raise ValueError(f"Expected embeddings list, got {type(embeddings)}")
        return embeddings
```

```python
async def retrieve_chunks(query: str, top_k: int = 6) -> list[dict]:
    """Retrieve relevant chunks using hybrid search (semantic + full-text)."""
    embedding = await ollama.generate_embedding(query)

    pool = await get_pool()
    async with pool.acquire() as conn:
        rows = await conn.fetch(

                content,
                source,
                metadata,
                similarity
            FROM hybrid_search_chunks($1::vector, $2, $3)
```

Modify the `build_messages()` function in `apps/backend/app/routes/chat.py` to inject retrieved context for `rag` and `coach` modes. For `normal` mode, behavior is unchanged.

```python
async def build_messages(
    session_id: str,
    user_message: str,
    mode: str,
) -> list[dict]:
    pool = await get_pool()
    repo = MessageRepository(pool)
    recent = await repo.get_recent(session_id, settings.chat_history_window)

    messages: list[dict] = []
    system_prompt = get_system_prompt(mode)
    messages.append({"role": "system", "content": system_prompt})

    # Add conversation history
    for msg in recent:
        messages.append({"role": msg["role"], "content": msg["content"]})

    # Inject RAG context for rag/coach modes
    if mode in ("rag", "coach"):
        from app.services.retrieval import retrieve_chunks
        chunks = await retrieve_chunks(user_message, top_k=settings.rag_top_k)
        if chunks:
            context_lines = ["---CONTEXT---"]
            for i, chunk in enumerate(chunks, 1):
                heading = chunk.get("metadata", {}).get("heading_path", "")
                source = chunk.get("source", "unknown")
                heading_str = f" > {heading}" if heading else ""
                context_lines.append(f"[{i}] From: {source}{heading_str}")
                context_lines.append(chunk["content"][:500])  # truncate for prompt safety
                context_lines.append("")
            context_lines.append("---CONTEXT---")
            context_block = "\n".join(context_lines)
            messages.append({"role": "user", "content": context_block})

    messages.append({"role": "user", "content": user_message})
    return messages
```

The `---CONTEXT---` block is a special user message that the model recognizes. It is prepended before the actual user message. Headings come from `chunk["metadata"].get("heading_path", "")`.
</action>

- For `mode="normal"`: behavior is identical to before (no retrieval, no context)
- For `mode="rag"` and `mode="coach"`: retrieves top_k chunks and prepends a context user message
- Context block is formatted as `---CONTEXT---\n[N] From: {source} > {heading}\n{text}\n...---CONTEXT---`
- If `retrieve_chunks` returns empty list, no context block is added (graceful degradation)
- `rag_top_k` comes from `settings.rag_top_k` (default 6)

```python
"""Debug search endpoint for RAG retrieval."""
from __future__ import annotations

import json
from fastapi import APIRouter, HTTPException

from app.models import SearchRequest
from app.services.retrieval import retrieve_chunks

router = APIRouter(prefix="/search", tags=["search"])


@router.post("")
async def search(request: SearchRequest) -> dict:
    """Search the knowledge base using hybrid retrieval.

    Returns top chunks with scores for debugging retrieval quality.
    """
    try:
        chunks = await retrieve_chunks(request.query, top_k=request.top_k)
        return {
            "query": request.query,
            "top_k": request.top_k,
            "chunks": [
                {
                    "id": c["id"],
                    "source": c["source"],
                    "heading": c.get("metadata", {}).get("heading_path", ""),
                    "text": c["content"],
                    "score": c["score"],
                }
                for c in chunks
            ],
            "count": len(chunks),
        }
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```
