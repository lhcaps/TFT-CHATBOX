Add a `CitationBadge` or inline citation list:

```tsx
function CitationList({ citations }: { citations: Citation[] }) {
  if (!citations?.length) return null;
  return (
    <div className="mt-2 flex flex-wrap gap-2">
      {citations.map((c, i) => (
        <span
          key={c.id || i}
          className="text-xs px-2 py-1 rounded bg-zinc-800 text-zinc-400 border border-zinc-700"
          title={c.text}
        >
          [{c.source.split(/[/\\]/).pop() || 'doc'}] {c.heading ? `> ${c.heading}` : ''}
        </span>
      ))}
    </div>
  );
}
```

In the message render for `role === 'assistant'`:

```tsx

</div>
```

  similarity: number;
}
