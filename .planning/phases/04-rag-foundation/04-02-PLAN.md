```python
def split_into_chunks(text: str, chunk_size: int = 2000, overlap: int = 500) -> Iterator[str]:
    """Split text into overlapping chunks of chunk_size characters, preserving paragraph boundaries.

    Steps:
    1. Split on double-newline boundaries (paragraphs) to keep coherent text together
    2. Accumulate paragraphs until reaching chunk_size
    3. If a single paragraph exceeds chunk_size, split on sentence boundaries
    4. Step back by `overlap` chars to create the next chunk
    """
    import re

    # Normalize whitespace
    text = re.sub(r'\r\n', '\n', text)
    text = re.sub(r'\n{3,}', '\n\n', text)

    # Split into paragraphs (preserve blank lines as paragraph breaks)
    paragraphs = re.split(r'\n\n+', text)
    paragraphs = [p.strip() for p in paragraphs if p.strip()]

    chunks: list[str] = []
    start = 0

    while start < len(text):
        end = min(start + chunk_size, len(text))

        # If not at end, try to break at paragraph or sentence boundary
        if end < len(text):
            # Try paragraph boundary first
            break_point = text.rfind('\n\n', start, end)
            if break_point > start:
                end = break_point + 2
            else:
                # Try sentence boundary
                break_point = max(
                    text.rfind('. ', start, end),
                    text.rfind('!\n', start, end),
                    text.rfind('?\n', start, end),
                )
                if break_point > start:
                    end = break_point + 2

        chunk = text[start:end].strip()
        if chunk:
            chunks.append(chunk)

        # Move start back by overlap for next iteration
        start = end - overlap
        if start <= chunks[-1] if chunks else start:
            start = end

    return iter(chunks)
```

Key changes:
- Character-based (not word-based) — word splitting caused uneven chunks
- Preserves paragraph boundaries where possible
- Falls back to sentence boundaries within paragraphs
- Falls back to hard break if paragraph itself exceeds chunk_size
- `overlap` is the step-back amount, not chunk_size - overlap
</action>

- Paragraphs shorter than chunk_size stay intact
- Function signature: `split_into_chunks(text: str, chunk_size: int = 2000, overlap: int = 500) -> Iterator[str]`
</acceptance_criteria>

from pathlib import Path
import asyncio

from app.db import get_pool
from app.services.ollama import ollama
from app.utils.markdown import extract_frontmatter, split_into_chunks
from app.utils.hashing import content_hash


async def _extract_heading_path(md_file: Path) -> str:
    """Extract the relative path as heading path."""
    return str(md_file.relative_to(md_file.anchor))


async def _split_with_heading(md_file: Path, body: str, chunk_size: int, overlap: int) -> list[tuple[str, dict]]:
    """Split markdown into chunks, each with heading metadata."""
    heading_path = str(md_file.relative_to(md_file.anchor))
    chunks = list(split_into_chunks(body, chunk_size=chunk_size, overlap=overlap))
    metadata = {"heading_path": heading_path, "source": str(md_file)}
    return [(chunk, metadata) for chunk in chunks]


async def ingest_vault(vault_path: str, chunk_size: int = 2000, overlap: int = 500) -> dict[str, int]:
    """Ingest all markdown files from an Obsidian vault."""
    vault = Path(vault_path)
    if not vault.exists():
        raise FileNotFoundError(f"Vault not found: {vault_path}")

    pool = await get_pool()
    stats = {"files": 0, "chunks": 0, "skipped": 0}

    BATCH_SIZE = 16  # VRAM-safe batch size for 16GB GPU

    async with pool.acquire() as conn:
        for md_file in vault.rglob("*.md"):
            # Skip .obsidian directory
            if ".obsidian" in md_file.parts:
                continue

            file_text = md_file.read_text(encoding="utf-8")
            file_hash = content_hash(file_text)

            # Check if file changed
            existing = await conn.fetchval(
                "SELECT 1 FROM chunks WHERE source = $1 AND content_hash = $2",
                str(md_file),
                file_hash,
            )
            if existing:
                stats["skipped"] += 1
                continue

            # Parse and split
            fm, body = extract_frontmatter(file_text)
            chunks_with_meta = await _split_with_heading(md_file, body, chunk_size, overlap)

            # Batch embed
            texts = [chunk for chunk, _ in chunks_with_meta]
            if texts:
                embeddings = await ollama.generate_embeddings(texts)

                # Bulk insert
                for (chunk_text, meta), embedding in zip(chunks_with_meta, embeddings):
                    chunk_hash = content_hash(chunk_text)
                    await conn.execute(
                        """
                        INSERT INTO chunks (content, content_hash, source, metadata, embedding)
                        VALUES ($1, $2, $3, $4, $5::vector)
                        ON CONFLICT (source, content_hash) DO NOTHING
                        """,
                        chunk_text,
                        chunk_hash,
                        str(md_file),
                        meta,
                        embedding,
                    )
                    stats["chunks"] += 1

            stats["files"] += 1

    return stats
```

from app.scripts.ingest_obsidian import ingest_vault

@router.post("/obsidian")
async def ingest_obsidian() -> dict:
    vault_path = settings.obsidian_vault_path
    if not vault_path:
        raise HTTPException(status_code=400, detail="OBSIDIAN_VAULT_PATH not set")
    if not Path(vault_path).exists():
        raise HTTPException(status_code=404, detail="Vault path not found")

    try:
        result = await ingest_vault(
            vault_path,
            chunk_size=settings.rag_chunk_size,  # now 2000
            overlap=settings.rag_chunk_overlap,  # now 500
        )
        return {"status": "ok", "result": result}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

Note: `ON CONFLICT (source, content_hash) DO NOTHING` handles re-ingest of changed files. The file hash is checked first; if unchanged, file is skipped entirely. For changed files, old chunks are NOT auto-deleted (they'll accumulate if the file shrinks). This is acceptable for MVP. The file hash check is the primary deduplication mechanism.
