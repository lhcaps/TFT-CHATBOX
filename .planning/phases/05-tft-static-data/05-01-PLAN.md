The file should implement these functions:

**CDN URLs (verified via web search):**
- Version list: `https://ddragon.leagueoflegends.com/api/versions.json` (not `/cdn/versions.json`)
- Champions: `https://ddragon.leagueoflegends.com/cdn/{version}/data/en_US/tft/tft-champion.json`
- Traits: `https://ddragon.leagueoflegends.com/cdn/{version}/data/en_US/tft/tft-trait.json`
- Items: `https://ddragon.leagueoflegends.com/cdn/{version}/data/en_US/tft/tft-item.json`
- Augments: `https://ddragon.leagueoflegends.com/cdn/{version}/data/en_US/tft/tft-augments.json`

**Cache structure:**
- Root: `settings.tft_cache_dir` (e.g. `~/.tft-copilot/cache/`)
- Per-patch folder: `~/.tft-copilot/cache/{patch}/`
- Files within: `champions.json`, `traits.json`, `items.json`, `augments.json`
- Version marker: `~/.tft-copilot/cache/{patch}/version.txt` (stores the CDN version string)

**Version check:**
1. Fetch `https://ddragon.leagueoflegends.com/api/versions.json` → parse JSON array → first entry = latest version
2. Check `~/.tft-copilot/cache/latest_version.txt` — if matches, cache is current, skip download
3. If mismatch, proceed with download and update `latest_version.txt`

**Per-patch cache logic (per D-05, D-06):**
- `get_patch_cache(patch: str)` → returns dict with keys: champions, traits, items, augments
  - Check if `~/.tft-copilot/cache/{patch}/` exists
  - If yes: read all 4 JSON files from disk, return dict
  - If no: download from CDN, save to disk, return dict
- Raise `FileNotFoundError` if patch folder does not exist AND CDN fetch fails

**Download function:**
- `fetch_json(url: str) -> dict` — async HTTP GET with httpx, raise on non-200
- Use `httpx.AsyncClient(timeout=30.0)` pattern matching existing code

**File structure:**
```python
from __future__ import annotations

import asyncio
import json
from pathlib import Path
from typing import TypedDict

import httpx

from app.config import settings


class TFTStaticData(TypedDict):
    champions: dict
    traits: dict
    items: dict
    augments: dict


# CDN base
DDAGON_BASE = "https://ddragon.leagueoflegends.com/cdn"
CDN_VERSIONS_URL = "https://ddragon.leagueoflegends.com/api/versions.json"
LOCAL_VERSION_FILE = Path(settings.tft_cache_dir) / "latest_version.txt"


async def fetch_json(url: str) -> dict:
    """Download JSON from URL."""
    async with httpx.AsyncClient(timeout=30.0) as client:
        response = await client.get(url)
        response.raise_for_status()
        return response.json()


async def get_latest_version() -> str:
    """Fetch the latest Data Dragon version from Riot CDN."""
    versions = await fetch_json(CDN_VERSIONS_URL)
    if not isinstance(versions, list) or not versions:
        raise ValueError(f"Invalid versions.json response: {versions}")
    return versions[0]


async def get_cached_version() -> str | None:
    """Read the cached version from disk, if any."""
    if not LOCAL_VERSION_FILE.exists():
        return None
    return LOCAL_VERSION_FILE.read_text(encoding="utf-8").strip()


async def save_cached_version(version: str) -> None:
    """Save the version string to disk."""
    LOCAL_VERSION_FILE.parent.mkdir(parents=True, exist_ok=True)
    LOCAL_VERSION_FILE.write_text(version, encoding="utf-8")


async def download_patch_data(version: str) -> TFTStaticData:
    """Download all TFT static data for a specific version from Riot CDN."""
    base = f"{DDAGON_BASE}/{version}/data/en_US/tft"

    champions, traits, items, augments = await asyncio.gather(
        fetch_json(f"{base}/tft-champion.json"),
        fetch_json(f"{base}/tft-trait.json"),
        fetch_json(f"{base}/tft-item.json"),
        fetch_json(f"{base}/tft-augments.json"),
    )

    return TFTStaticData(
        champions=champions,
        traits=traits,
        items=items,
        augments=augments,
    )


def get_patch_cache_dir(patch: str) -> Path:
    """Return the cache directory for a specific patch version."""
    return Path(settings.tft_cache_dir) / patch


async def save_patch_to_cache(patch: str, data: TFTStaticData, version: str) -> None:
    """Save downloaded patch data to disk cache."""
    cache_dir = get_patch_cache_dir(patch)
    cache_dir.mkdir(parents=True, exist_ok=True)

    for name, content in data.items():
        (cache_dir / f"{name}.json").write_text(json.dumps(content, ensure_ascii=False), encoding="utf-8")

    # Save version marker
    (cache_dir / "version.txt").write_text(version, encoding="utf-8")


async def load_patch_from_cache(patch: str) -> TFTStaticData | None:
    """Load patch data from disk cache. Returns None if cache miss."""
    cache_dir = get_patch_cache_dir(patch)
    if not cache_dir.exists():
        return None

    result: TFTStaticData = {}
    for name in ("champions", "traits", "items", "augments"):
        file_path = cache_dir / f"{name}.json"
        if not file_path.exists():
            return None  # Incomplete cache
        result[name] = json.loads(file_path.read_text(encoding="utf-8"))

    return result


async def get_patch_data(patch: str | None = None) -> tuple[str, TFTStaticData]:
    """Get TFT static data for a patch, using cache when available.

    If patch is None, fetches latest version from Riot CDN.
    Returns (patch_version, data_dict).
    """
    if patch is None:
        patch = await get_latest_version()

    # Try cache first
    cached = await load_patch_from_cache(patch)
    if cached is not None:
        return patch, cached

    # Download from CDN
    version, data = await get_latest_version(), None
    if patch == version:
        data = await download_patch_data(version)
        await save_patch_to_cache(version, data, version)
    else:
        # Specific old patch — fetch directly
        data = await download_patch_data(patch)

    return patch, data


async def ingest_tft_static(patch: str | None = None) -> dict[str, int]:
    """Download and cache TFT static data from Riot CDN.

    Checks local cache before downloading. Stores downloaded data
    in ~/.tft-copilot/cache/{patch}/ for reuse.
    """
    version, _ = await get_patch_data(patch)
    # Check if we need to update global version marker
    cached_version = await get_cached_version()
    if version != cached_version:
        await save_cached_version(version)

    # Return stats (actual parsing/chunking done in Plan 05-02)
    return {"patch": version, "status": "downloaded", "types": 4}
```

**Key design points:**
- `get_patch_data()` handles both cache-hit and cache-miss — returns (patch_version, data_dict)
- When `patch=None`: fetches latest version, checks cache, saves to cache on miss
- When `patch="17.1"`: tries cache first, falls back to CDN (fetches specific version from CDN if not cached)
- `ingest_tft_static()` is a thin wrapper that returns download stats (chunking happens separately in Plan 05-02)
- Uses `asyncio.gather()` for parallel download of all 4 data types (faster)
- HTTP errors raise via `response.raise_for_status()` — caller handles exceptions
