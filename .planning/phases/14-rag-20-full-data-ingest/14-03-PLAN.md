Then during token streaming (inside the `async for line in response.aiter_lines():` loop), add citation_progress logic:

```python
# Inside async for line in response.aiter_lines():
async for line in response.aiter_lines():
    if not line.strip():
        continue
    try:
        chunk_data = json.loads(line)
        content = chunk_data.get("message", {}).get("content", "")
        
        if content:
            full_text += content
            
            # citation_progress: reveal citation content based on token position
            # Simple heuristic: if a citation ID marker appears in the token stream,
            # reveal that citation's content
            for citation_id, cit in streaming_citations.items():
                if not cit["revealed"] and citation_id in streaming_citations:
                    # Progressive reveal: first 100 chars of content preview
                    preview = chunk["content"][:100] if chunk else ""
                    yield f"event: citation_progress\ndata: {json.dumps({'citation': {
                        'id': citation_id,
                        'text_preview': preview,
                    })})\n\n"
                    cit["revealed"] = True
                    break
            
            yield f"event: token\ndata: {json.dumps({'token': content})}\n\n"
```

```tsx
function StreamingCitationCard({ citation }: { citation: StreamingCitation }) {
  const isLoading = citation.status === 'loading';
  const isProgress = citation.status === 'progress';
  
  return (
    <div className="border border-gray-700 rounded-xl p-3 bg-gray-800/50">
      <div className="flex items-center justify-between gap-2 mb-1">
        <span className="text-xs font-mono text-gray-400 truncate flex-1">
          {citation.source}
        </span>
        {citation.status === 'complete' && (
          <span className="text-xs text-gray-500">
            {Math.round(citation.score * 100)}%
          </span>
        )}
        {citation.status === 'loading' && (
          <span className="text-xs text-purple-400 animate-pulse">
            Loading...
          </span>
        )}
      </div>
      {citation.heading && (
        <div className="text-xs text-gray-500 truncate mb-1">
          {citation.heading}
        </div>
      )}
      {isLoading && (
        <div className="space-y-1">
          <div className="h-3 bg-gray-700 rounded animate-pulse w-full" />
          <div className="h-3 bg-gray-700 rounded animate-pulse w-3/4" />
        </div>
      )}
      {isProgress && (
        <p className="text-sm text-gray-400 italic">
          {citation.text_preview || '...'}
        </p>
      )}
      {citation.status === 'complete' && (
        <p className="text-sm text-gray-300">
          {citation.text.length > 200 
            ? citation.text.slice(0, 200) + '...'
            : citation.text}
        </p>
      )}
    </div>
  );
}
```
