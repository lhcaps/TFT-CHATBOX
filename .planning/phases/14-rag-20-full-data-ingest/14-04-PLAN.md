"""Base processor class for modular data ingestion."""
from __future__ import annotations

import hashlib
import json
import logging
from abc import ABC, abstractmethod
from pathlib import Path
from typing import Any

import httpx

from app.config import settings
from app.db import get_pool
from app.services.ollama import ollama

logger = logging.getLogger(__name__)


class BaseProcessor(ABC):
    """Base class for all data processors."""
    
    # Override in subclass
    source_file: str = ""  # JSON filename in project root
    entity_type: str = ""  # entity_type metadata value
    chunk_size: int = 300  # target chars per chunk
    
    def __init__(self):
        self.ingested_count = 0
        self.skipped_count = 0
        self.seen_hashes: set[str] = set()
    
    @abstractmethod
    def process_item(self, item: dict[str, Any]) -> list[dict[str, Any]]:
        """Convert a single data item into chunk(s).
        
        Returns list of chunk dicts with: content, metadata
        """
        pass
    
    def get_data_path(self) -> Path:
        """Get path to data file in project root."""
        # Project root is 2 levels up from scripts/ingest/
        backend_dir = Path(__file__).parent.parent.parent
        return backend_dir / self.source_file
    
    def load_data(self) -> list[dict[str, Any]]:
        """Load JSON data from source file."""
        data_path = self.get_data_path()
        if not data_path.exists():
            raise FileNotFoundError(f"Data file not found: {data_path}")
        with open(data_path, "r", encoding="utf-8") as f:
            return json.load(f)
    
    def compute_hash(self, content: str) -> str:
        """Compute content hash for deduplication."""
        return hashlib.md5(content.encode()).hexdigest()
    
    async def ingest(self) -> dict[str, Any]:
        """Main ingest method — loads data, processes items, upserts chunks."""
        logger.info(f"Starting {self.__class__.__name__} ingest...")
        
        data = self.load_data()
        all_chunks: list[dict[str, Any]] = []
        
        for item in data:
            try:
                chunks = self.process_item(item)
                for chunk in chunks:
                    # Hash-based dedup
                    content_hash = self.compute_hash(chunk["content"])
                    if content_hash in self.seen_hashes:
                        self.skipped_count += 1
                        continue
                    self.seen_hashes.add(content_hash)
                    all_chunks.append(chunk)
            except Exception as e:
                logger.warning(f"Failed to process item: {e}")
                self.skipped_count += 1
        
        # Generate embeddings and upsert to DB
        if all_chunks:
            await self._upsert_chunks(all_chunks)
        
        logger.info(f"{self.__class__.__name__}: ingested={self.ingested_count}, skipped={self.skipped_count}")
        return {
            "processor": self.__class__.__name__,
            "ingested": self.ingested_count,
            "skipped": self.skipped_count,
            "total_chunks": len(all_chunks),
        }
    
    async def _upsert_chunks(self, chunks: list[dict[str, Any]]) -> None:
        """Generate embeddings and upsert chunks to database."""
        pool = await get_pool()
        
        for chunk in chunks:
            # Generate embedding
            raw_embedding = await ollama.generate_embedding(chunk["content"])
            embedding = raw_embedding[: settings.ollama_embedding_dims]
            
            # Upsert to DB
            async with pool.acquire() as conn:
                await conn.execute(
                    """
                    INSERT INTO document_chunks (content, embedding, source, metadata)
                    VALUES ($1, $2::vector, $3, $4)
                    ON CONFLICT DO NOTHING
                    """,
                    chunk["content"],
                    json.dumps(embedding),
                    chunk.get("source", self.source_file),
                    json.dumps(chunk.get("metadata", {})),
                )
            self.ingested_count += 1

"""Processor for augments_full_user_verified.json — 252 augments."""
from __future__ import annotations

from .base_processor import BaseProcessor


class AugmentProcessor(BaseProcessor):
    source_file = "augments_full_user_verified.json"
    entity_type = "augment"
    chunk_size = 350  # ~200-500 chars per augment
    
    def process_item(self, item: dict) -> list[dict]:
        """Convert augment to chunk.
        
        Each augment: {name, tier, description, relevant_round?} 
        Chunks are ~350 chars.
        """
        name = item.get("name", "")
        tier = item.get("tier", "Unknown")  # Silver, Gold, Prismatic
        description = item.get("description", "")
        relevant_round = item.get("relevant_round", "")
        
        # Build content
        content_parts = [f"## {name} ({tier} Augment)"]
        if relevant_round:
            content_parts.append(f"Best for: {relevant_round}")
        content_parts.append(description)
        content = "\n\n".join(content_parts)
        
        # Truncate to chunk_size
        if len(content) > self.chunk_size:
            content = content[: self.chunk_size - 3] + "..."
        
        return [{
            "content": content,
            "source": self.source_file,
            "metadata": {
                "entity_type": self.entity_type,
                "augment_name": name,
                "tier": tier,
                "relevant_round": relevant_round,
                "patch": "17.1",
                "set": 17,
                "ingested_at": __import__("datetime").datetime.now().isoformat(),
            }
        }]

from __future__ import annotations

from .base_processor import BaseProcessor


class ChampionProcessor(BaseProcessor):
    source_file = "tft_set17_patch17_1_deep_pack_v4_user_verified.json"
    entity_type = "champion"
    chunk_size = 400
    
    def process_item(self, item: dict) -> list[dict]:
        """Convert champion to chunk.
        
        Item format: {name, cost, traits_numeric: {origins, classes}, ability_summary, role, items_ideal?}
        """
        name = item.get("name", "")
        cost = item.get("cost", 0)
        traits = item.get("traits_numeric", {})
        origins = traits.get("origins", [])
        classes = traits.get("classes", [])
        ability = item.get("ability_summary", "")
        role = item.get("role", "")
        
        # Format traits
        trait_list = []
        for origin in origins:
            trait_list.append(f"{origin['name']} ({origin.get('count', 1)})")
        for cls in classes:
            trait_list.append(f"{cls['name']} ({cls.get('count', 1)})")
        traits_str = ", ".join(trait_list) if trait_list else "No traits"
        
        content = f"""## {name} (Cost {cost})

**Traits:** {traits_str}
**Role:** {role}
**Ability:** {ability}
"""
        
        # Truncate to chunk_size
        if len(content) > self.chunk_size:
            content = content[: self.chunk_size - 3] + "..."
        
        return [{
            "content": content,
            "source": self.source_file,
            "metadata": {
                "entity_type": self.entity_type,
                "champion_name": name,
                "cost": cost,
                "traits": [t['name'] for t in origins] + [t['name'] for t in classes],
                "role": role,
                "patch": "17.1",
                "set": 17,
                "ingested_at": __import__("datetime").datetime.now().isoformat(),

        # Truncate

            content = content[: self.chunk_size - 3] + "..."

            "content": content,
            "source": self.source_file,

                "entity_type": self.entity_type,

class RollingOddsProcessor(BaseProcessor):
    source_file = "rolling_odds_user_verified.json"
    entity_type = "system"
    chunk_size = 200  # Compact format
    
    def process_item(self, item: dict) -> list[dict]:
        """Convert rolling odds to chunk.
        
        Rolling odds table by level.
        """
        # Handle different possible formats
        level = item.get("level", item.get("dvo_level", ""))
        odds = item.get("odds", item.get("rolling_odds", {}))
        
        content_parts = [f"## Rolling Odds — Level {level}"]

        if isinstance(odds, dict):

        return [{
            "content": content,
            "source": self.source_file,

from __future__ import annotations

from .base_processor import BaseProcessor


class DataPackProcessor(BaseProcessor):
    source_file = "tft_set17_patch17_1_data_pack.json"
    entity_type = "champion"  # primarily champion data
    chunk_size = 400
    
    def process_item(self, item: dict) -> list[dict]:
        """Convert data pack item to chunk.
        
        Data pack has: champions (with abilities), systems, etc.
        Determine entity_type from item structure.
        """
        item_type = item.get("type", "")
        
        if item_type == "champion" or "ability" in item:
            return self._process_champion(item)
        elif item_type == "system":
            return self._process_system(item)
        else:
            return self._process_generic(item)
    
    def _process_champion(self, item: dict) -> list[dict]:
        name = item.get("name", "")
        cost = item.get("cost", 0)
        ability = item.get("ability", {})
        ability_name = ability.get("name", "") if isinstance(ability, dict) else ""
        ability_desc = ability.get("description", "") if isinstance(ability, dict) else ""
        
        content = f"""## {name} (Cost {cost})

**Ability:** {ability_name}
{ability_desc}
"""
        if len(content) > self.chunk_size:
            content = content[: self.chunk_size - 3] + "..."
        
        return [{
            "content": content,
            "source": self.source_file,
            "metadata": {
                "entity_type": "champion",
                "champion_name": name,
                "cost": cost,
                "patch": "17.1",
                "set": 17,
                "ingested_at": __import__("datetime").datetime.now().isoformat(),
            }
        }]
    
    def _process_system(self, item: dict) -> list[dict]:
        name = item.get("name", "")
        description = item.get("description", "")
        
        content = f"""## {name} (System)

{description}
"""
        if len(content) > self.chunk_size:
            content = content[: self.chunk_size - 3] + "..."
        
        return [{
            "content": content,
            "source": self.source_file,
            "metadata": {
                "entity_type": "system",
                "system_name": name,
                "patch": "17.1",
                "set": 17,
                "ingested_at": __import__("datetime").datetime.now().isoformat(),
            }
        }]
    
    def _process_generic(self, item: dict) -> list[dict]:
        name = item.get("name", str(item)[:100])  # Fallback
        content = f"## {name}\n\n{item.get('description', str(item))}"
        
        if len(content) > self.chunk_size:
            content = content[: self.chunk_size - 3] + "..."
        
        return [{
            "content": content,
            "source": self.source_file,
            "metadata": {
                "entity_type": "general",
                "patch": "17.1",
                "set": 17,
                "ingested_at": __import__("datetime").datetime.now().isoformat(),
            }
        }]

class BatchIngestRequest(BaseModel):
    """Request model for batch ingest endpoint."""

@router.post("/batch")
async def ingest_batch(request: BatchIngestRequest) -> dict:
    """Trigger batch ingestion from a specific data source.
    
    Uses modular processor classes to ingest TFT data into vector DB.
    Each processor handles one data file with entity_type metadata.
    
    Available processors:
    - augment: augments_full_user_verified.json (252 augments)
    - champion: tft_set17_patch17_1_deep_pack_v4_user_verified.json (59 champions)
    - item: items_effects_expanded_set17.json (45+ items)
    - trait: traits_full_user_verified.json (30+ traits)
    - system: tft_set17_patch17_1_enhanced_pack.json (9 Space Gods)
    - rolling_odds: rolling_odds_user_verified.json
    - data_pack: tft_set17_patch17_1_data_pack.json (full roster)
    
    All processors add entity_type metadata and target ≥500 total chunks.
    """
    from scripts.ingest import PROCESSOR_MAP
    
    processor_name = request.processor.lower()
    
    if processor_name not in PROCESSOR_MAP:
        raise HTTPException(
            status_code=400,
            detail=f"Unknown processor: {processor_name}. Available: {list(PROCESSOR_MAP.keys())}"
        )
    
    processor_class = PROCESSOR_MAP[processor_name]
    processor = processor_class()
    
    try:
        result = await processor.ingest()
        trigger_reload()  # Reload knowledge graph after ingest
        return {"status": "ok", **result}
    except FileNotFoundError as e:
        raise HTTPException(status_code=404, detail=str(e))
    except Exception as e:
        logger.exception(f"Batch ingest failed for {processor_name}")
        raise HTTPException(status_code=500, detail=f"Ingest failed: {e}")

    Target: ≥500 total chunks with entity_type metadata.
    """
    from scripts.ingest import PROCESSOR_MAP
    
    results = {}
    total_ingested = 0
    total_skipped = 0
    
    for name, processor_class in PROCESSOR_MAP.items():
        try:
            processor = processor_class()
            result = await processor.ingest()
            results[name] = result
            total_ingested += result.get("ingested", 0)
            total_skipped += result.get("skipped", 0)
        except Exception as e:
            logger.exception(f"Failed {name} processor")
            results[name] = {"error": str(e)}
    
    trigger_reload()
    
    return {
        "status": "ok",
        "results": results,
        "total_ingested": total_ingested,
        "total_skipped": total_skipped,
    }
