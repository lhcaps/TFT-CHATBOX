```python
"""Knowledge Graph module — NetworkX MultiDiGraph for TFT entities."""
from functools import lru_cache
from networkx import MultiDiGraph
import logging

logger = logging.getLogger(__name__)

class LazyGraphLoader:
    """Thread-safe lazy-loading wrapper around nx.MultiDiGraph.
    
    Defers graph construction until first access.
    Call reload() to rebuild from JSON files.
    """
    _instance: "LazyGraphLoader | None" = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._graph: MultiDiGraph | None = None
            cls._instance._loaded = False
        return cls._instance

    def get(self) -> MultiDiGraph:
        if self._graph is None:
            from app.graph.builder import build_graph
            self._graph = build_graph()
            self._loaded = True
            logger.info(f"Knowledge graph loaded: {self._graph.number_of_nodes()} nodes, "
                       f"{self._graph.number_of_edges()} edges")
        return self._graph

    def reload(self) -> MultiDiGraph:
        """Clear and rebuild the graph from JSON sources."""
        logger.info("Reloading knowledge graph...")
        self._graph = None
        self._loaded = False
        return self.get()

    @property
    def is_loaded(self) -> bool:
        return self._loaded

    @property
    def node_counts(self) -> dict[str, int]:
        """Return node counts by type."""
        from collections import Counter
        G = self.get()
        return dict(Counter(
            G.nodes[n].get("type", "unknown") for n in G.nodes()
        ))

# Singleton instance — import this instead of LazyGraphLoader()
knowledge_graph = LazyGraphLoader()
```

```python
"""Centralized JSON loading utilities for graph construction."""
import json
from pathlib import Path

def get_data_dir() -> Path:
    """Return the project root data directory (parent of backend/)."""
    return Path(__file__).parent.parent.parent.parent / "data"

def load_json(filename: str) -> dict:
    """Load a JSON file from the data directory.
    
    Args:
        filename: Name of JSON file (e.g., "augments_full_user_verified.json")
    Returns:
        Parsed JSON dict
    Raises:
        FileNotFoundError: If file doesn't exist
    """
    data_dir = get_data_dir()
    file_path = data_dir / filename
    with open(file_path, encoding="utf-8") as f:
        return json.load(f)
```

TRAIT_ROLE_MAP = {
    "anima": "TANK", "primordian": "TANK", "vanguard": "TANK", "bastion": "TANK",
    "invoker": "AP", "conduit": "AP",
    "rogue": "AD", "marauder": "AD", "sniper": "AD", "challenger": "AD",
}

def infer_role(cost: int, traits: list) -> str:
    """Infer champion role from traits first, then cost."""
    trait_names = [t.lower() for t in traits]
    for trait in trait_names:
        if trait in TRAIT_ROLE_MAP:
            return TRAIT_ROLE_MAP[trait]
    if cost >= 4:
        return "AD"
    return "TANK"

def categorize_item(recipe: list[str]) -> str:
    """Categorize item by its components.
    B.F. Sword + Giant's Belt → AD
    Needlessly Large Rod + Giant's Belt → AP
    Chain Vest + Negatron Cloak → TANK
    etc.
    """
    # Component categories
    AD_COMPONENTS = {"b_f_sword", "recurve_bow"}
    AP_COMPONENTS = {"needlessly_large_rod", "tear_of_the_goddess"}
    TANK_COMPONENTS = {"chain_vest", "giants_belt", "negatron_cloak"}

    normalized_recipe = {re.sub(r'[^a-z0-9]+', '_', r.lower().strip()).strip('_') for r in recipe}
    
    if normalized_recipe & AD_COMPONENTS and not (normalized_recipe & AP_COMPONENTS):
        return "AD"
    if normalized_recipe & AP_COMPONENTS:
        return "AP"
    if normalized_recipe & TANK_COMPONENTS and len(normalized_recipe - TANK_COMPONENTS) <= 1:
        return "TANK"
    return "SUPPORT"

def assign_item_edges(G: MultiDiGraph, champions: list, items: list) -> None:
    """Assign ITEM_FOR_CHAMPION edges based on inferred role.
    
    For each champion, assign top 3 items matching their role.
    """
    # Build item index by category
    items_by_role = {"AD": [], "AP": [], "TANK": [], "SUPPORT": []}
    for item in items:
        item_id = normalize_id(item["name"])
        if item_id not in G:
            continue
        role = categorize_item(item.get("recipe", []))
        items_by_role[role].append(item_id)

    # Assign to champions
    for champ in champions:
        champ_id = normalize_id(champ["name"])
        if champ_id not in G:
            continue
        role = infer_role(champ.get("cost", 3), champ.get("traits", []))
        for item_id in items_by_role.get(role, [])[:3]:
            G.add_edge(item_id, champ_id,
                edge_type="ITEM_FOR_CHAMPION",
                role=role,
                confidence="heuristic",
            )

```python
"""Unit tests for knowledge graph — KNOW-04 coverage."""
import pytest
from app.graph import knowledge_graph
from app.graph.builder import build_graph, normalize_id

class TestKnowledgeGraph:
    def test_graph_loads_all_node_types(self):
        G = build_graph()
        types = {G.nodes[n].get("type") for n in G.nodes()}
        assert "champion" in types
        assert "item" in types
        assert "trait" in types
        assert "augment" in types

    def test_graph_minimum_node_counts(self):
        G = build_graph()
        from collections import Counter
        counts = Counter(G.nodes[n].get("type") for n in G.nodes())
        assert counts["champion"] >= 59, f"Expected >=59 champions, got {counts['champion']}"
        assert counts["trait"] >= 30, f"Expected >=30 traits, got {counts['trait']}"
        assert counts["item"] >= 45, f"Expected >=45 items, got {counts['item']}"
        assert counts["augment"] >= 252, f"Expected >=252 augments, got {counts['augment']}"

    def test_has_trait_edges(self):
        G = build_graph()
        briar_id = normalize_id("Briar")
        anima_id = normalize_id("Anima")
        assert anima_id in G.neighbors(briar_id), "Briar should have Anima trait"
        edge_data = G.get_edge_data(briar_id, anima_id)
        has_trait = any(
            e.get("edge_type") == "HAS_TRAIT" for e in edge_data.values()
        )
        assert has_trait, "Edge should be HAS_TRAIT"

    def test_builds_from_edges(self):
        G = build_graph()
        bt_id = normalize_id("Bloodthirster")
        bfs_id = normalize_id("B.F. Sword")
        assert bfs_id in G.neighbors(bt_id), "Bloodthirster should have B.F. Sword in recipe"

    def test_missing_entity_returns_404_format(self):
        from app.graph.models import ErrorResponse
        # Test that suggestions field exists in ErrorResponse
        err = ErrorResponse(error="not found", entity="fake", suggestions=["a", "b"])
        assert len(err.suggestions) == 2

    def test_graph_reload_preserves_structure(self):
        G1 = knowledge_graph.get()
        counts1 = knowledge_graph.node_counts
        G2 = knowledge_graph.reload()
        counts2 = knowledge_graph.node_counts
        assert counts1 == counts2, "Reload should preserve node counts"

    def test_circular_references_handled(self):
        G = build_graph()
        # Champions can have traits, items can be built from other items (no cycles expected)
        # Just verify graph is valid
        assert G.number_of_nodes() > 0

    def test_empty_champion_traits_handled(self):
        G = build_graph()
        # Any champion node should have the node, even if traits are empty
        for node in G.nodes():
            if G.nodes[node].get("type") == "champion":
                assert G.nodes[node].get("name") is not None

    def test_depth_3_traversal_no_infinite_loop(self):
        G = build_graph()
        import networkx as nx
        briar_id = normalize_id("Briar")
        # Depth-3 traversal should complete without hanging
        paths = list(nx.single_source_shortest_path_length(G, briar_id, cutoff=3))
        assert len(paths) > 0
```
